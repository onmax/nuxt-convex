---
title: File Storage
description: Upload files to Convex storage and display them with reactive URL subscriptions using the storage plugin.
---

File storage requires two plugins: `convexVue` for the base connection and `convexVueStorage` to register the backend function references that the storage composables call.

## Backend functions

Create the three Convex functions that the storage plugin expects.

```ts [convex/storage.ts]
import { v } from 'convex/values'
import { mutation, query } from './_generated/server'

export const generateUploadUrl = mutation(async (ctx) => {
  return await ctx.storage.generateUploadUrl()
})

export const getUrl = query({
  args: { storageId: v.id('_storage') },
  handler: async (ctx, { storageId }) => {
    return await ctx.storage.getUrl(storageId)
  },
})

export const remove = mutation({
  args: { storageId: v.id('_storage') },
  handler: async (ctx, { storageId }) => {
    await ctx.storage.delete(storageId)
  },
})
```

## Register the storage plugin

```ts [src/main.ts]
import { convexVue } from '@onmax/convex-vue'
import { convexVueStorage } from '@onmax/convex-vue/storage'
import { createApp } from 'vue'
import { api } from '../convex/_generated/api'
import App from './App.vue'

const app = createApp(App)

app.use(convexVue, {
  url: import.meta.env.VITE_CONVEX_URL,
})

app.use(convexVueStorage, {
  generateUploadUrl: api.storage.generateUploadUrl,
  getUrl: api.storage.getUrl,
  remove: api.storage.remove,
})

app.mount('#app')
```

## `useConvexStorage`

Access the low-level storage helpers directly.

```ts
import { useConvexStorage } from '@onmax/convex-vue/storage'

const storage = useConvexStorage()
```

::field-group
::field{name="generateUploadUrl" type="() => Promise<string>"}
Call the registered mutation to get a presigned upload URL.
::

::field{name="getUrl" type="(storageId: string) => Ref<string | null>"}
Subscribe to the public URL for a stored file. The ref updates in realtime.
::

::field{name="remove" type="(storageId: string) => Promise<void>"}
Delete a file from storage by its ID.
::
::

## `useConvexUpload`

A higher-level composable that handles the upload flow and exposes progress tracking.

```vue [src/components/FileUpload.vue]
<script setup lang="ts">
import { useConvexUpload } from '@onmax/convex-vue/storage'

const { upload, isUploading, progress, error } = useConvexUpload({
  onSuccess(storageId, file) {
    console.log(`Uploaded ${file.name} as ${storageId}`)
  },
})

async function onSelect(event: Event) {
  const file = (event.target as HTMLInputElement).files?.[0]
  if (file)
    await upload(file)
}
</script>

<template>
  <input type="file" @change="onSelect">
  <p v-if="isUploading">
    Uploading... {{ progress }}%
  </p>
  <p v-if="error">
    {{ error.message }}
  </p>
</template>
```

### Return values

::field-group
::field{name="upload" type="(file: File) => Promise<string | null>"}
Upload a file and return its `storageId`, or `null` on failure.
::

::field{name="isUploading" type="Ref<boolean>"}
`true` while a file upload is in progress.
::

::field{name="progress" type="Ref<number>"}
Upload progress from `0` to `100`.
::

::field{name="error" type="Ref<Error | null>"}
The latest upload error.
::
::

### Options

::field-group
::field{name="generateUploadUrl" type="() => Promise<string>"}
Override the default upload URL generator from the storage plugin.
::

::field{name="onSuccess" type="(storageId: string, file: File) => void"}
Called after a successful upload.
::

::field{name="onError" type="(error: Error) => void"}
Called when the upload fails.
::
::

::warning
File uploads only work on the client. Calling `upload()` during SSR sets an error and returns `null`.
::

## Display an uploaded file

Combine `useConvexUpload` with `useConvexStorage` to show a preview after uploading.

```vue [src/components/AvatarUpload.vue]
<script setup lang="ts">
import { useConvexStorage, useConvexUpload } from '@onmax/convex-vue/storage'

const storage = useConvexStorage()
const storageId = ref<string | null>(null)
const url = computed(() => storageId.value ? storage.getUrl(storageId.value) : null)

const { upload, isUploading } = useConvexUpload({
  onSuccess(id) {
    storageId.value = id
  },
})

async function onSelect(event: Event) {
  const file = (event.target as HTMLInputElement).files?.[0]
  if (file)
    await upload(file)
}
</script>

<template>
  <input type="file" accept="image/*" @change="onSelect">
  <img v-if="url?.value" :src="url.value" alt="Avatar preview">
  <p v-if="isUploading">
    Uploading...
  </p>
</template>
```
