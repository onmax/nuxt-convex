---
title: Cloudflare R2
description: Enable the Nuxt-side R2 upload helper and combine it with the Convex R2 component for bucket uploads.
---

Set `convex.r2` to `true` when you want the Nuxt module to auto-import `useConvexR2Upload`.

```ts [nuxt.config.ts]
export default defineNuxtConfig({
  modules: ['nuxt-convex'],
  convex: {
    r2: true,
  },
})
```

## What the module adds

When R2 support is enabled, `nuxt-convex`:

- auto-imports `useConvexR2Upload`
- keeps the upload flow inside the Nuxt runtime boundary
- does not create an R2-specific alias

## Install the Convex component

```bash
pnpm add @convex-dev/r2
```

```ts [convex/convex.config.ts]
import r2 from '@convex-dev/r2/convex.config.js'
import { defineApp } from 'convex/server'

const app = defineApp()
app.use(r2)

export default app
```

## Create the client API

```ts [convex/r2.ts]
import type { DataModel } from './_generated/dataModel'
import { R2 } from '@convex-dev/r2'
import { components } from './_generated/api'

export const r2 = new R2(components.r2)

export const { generateUploadUrl, syncMetadata } = r2.clientApi<DataModel>({
  checkUpload: async (_ctx, _bucket) => {},
  onUpload: async (_ctx, _bucket, _key) => {},
})
```

## Upload from Nuxt

```vue [app/components/R2Upload.vue]
<script setup lang="ts">
import { api } from '#convex/api'

const { upload, progress, isUploading, error } = useConvexR2Upload(api.r2)

async function onSelect(event: Event) {
  const file = (event.target as HTMLInputElement).files?.[0]
  if (file)
    await upload(file)
}
</script>
```
