---
title: useConvexMutation
description: Call Convex mutations from Vue or Nuxt and optionally pass through an optimistic update.
---

`useConvexMutation` wraps `client.mutation(...)` and tracks pending state and errors.

```vue [app/components/AddTask.vue]
<script setup lang="ts">
import { api } from '#convex/api'

const { mutate, isPending, error } = useConvexMutation(api.tasks.add)
const title = ref('')

async function submit() {
  await mutate({
    title: title.value,
    userId: 'user_123',
  })

  title.value = ''
}
</script>
```

## Signature

```ts
function useConvexMutation<Mutation extends FunctionReference<'mutation'>>(
  mutation: Mutation,
  options?: {
    optimisticUpdate?: OptimisticUpdate<FunctionArgs<Mutation>>
  },
): {
  mutate: (args: MaybeRefOrGetter<FunctionArgs<Mutation>>) => Promise<FunctionReturnType<Mutation>>
  error: Ref<Error | null>
  isPending: ComputedRef<boolean>
}
```

## Optimistic updates

The `optimisticUpdate` option is forwarded to the underlying Convex client.

```ts [app/components/AddTask.vue]
const { mutate } = useConvexMutation(api.tasks.add, {
  optimisticUpdate: (localStore, args) => {
    const current = localStore.getQuery(api.tasks.list, { userId: args.userId }) ?? []

    localStore.setQuery(api.tasks.list, { userId: args.userId }, [
      ...current,
      {
        _id: `optimistic-${Date.now()}` as never,
        _creationTime: Date.now(),
        title: args.title,
        userId: args.userId,
        isCompleted: false,
        createdAt: Date.now(),
      },
    ])
  },
})
```

Use this when you want the UI to update before the server round-trip finishes.
